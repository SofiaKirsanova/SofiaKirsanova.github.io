\begin{rSection}{Research Experience}

{\bf Graduate Research Assistant}, \textit{Knowledge Computing Lab, University of Minnesota, USA} \hfill 2024--Present

\textit{IMOLA (NSF). Automated Road Network Extraction \& Topological Mapping (ongoing)}  
\begin{itemize}
\item Develop deep-learning methods for extracting road networks from maps, with a focus on detecting road centerlines, intersections, bridges, and other linear transportation features across varying cartographic styles.
\item Build algorithms for topology reconstruction, converting raw line predictions into routable graph structures with nodes and edges classification and connectivity correction.
\item Design evaluation pipelines to assess extraction quality across large heterogeneous map collections with patch difficulty modeling, structured error attribution, and cross-region generalization metrics.
\end{itemize}

\textit{CriticalMAAS (DARPA). Legend Detection \& Map Layout Analysis (completed)}  
\begin{itemize}
\item Developed deep-learning methods for detecting legend symbols and text regions and linking them into structured legend items across diverse cartographic styles.
\item Combined fine-tuned LayoutLMv3 with GPT-4o structured prompting for layout-aware legend parsing in the \textbf{DIGMAPPER} map-digitization pipeline.
\item Achieved \textbf{96\%} symbol-detection F1, \textbf{97.8\%} text-detection F1, and \textbf{97.1\%} symbol–description linking accuracy.
\end{itemize}
